}

/**
 * src/hooks/useCases.js
 * ---------------------
 * Fetches and auto-refreshes the case list from the backend API.
 * Polling interval: 30 seconds (configurable via POLL_MS).
 */

import { useState, useEffect, useCallback } from 'react'
import { fetchCases, updateCase } from '../api/client'

const POLL_MS = 30_000

export function useCases() {
  const [cases, setCases]     = useState([])
  const [loading, setLoading] = useState(true)
  const [error, setError]     = useState(null)
  const [lastSync, setLastSync] = useState(null)

  const load = useCallback(async () => {
    try {
      const data = await fetchCases()
      setCases(data)
      setError(null)
      setLastSync(new Date())
    } catch (err) {
      setError(err.message)
    } finally {
      setLoading(false)
    }
  }, [])

  // Initial load + polling
  useEffect(() => {
    load()
    const id = setInterval(load, POLL_MS)
    return () => clearInterval(id)
  }, [load])

  const updateStatus = useCallback(async (caseId, status, notes = '', etaMinutes = null) => {
    await updateCase(caseId, { status, notes, eta_minutes: etaMinutes })
    // Optimistic local update
    setCases(prev =>
      prev.map(c => c.id === caseId ? { ...c, status, notes, eta_minutes: etaMinutes } : c)
    )
  }, [])

  const activeCases = cases.filter(c => !['COMPLETED', 'CANCELLED'].includes(c.status))
  const completedCases = cases.filter(c => ['COMPLETED', 'CANCELLED'].includes(c.status))

  return { cases, activeCases, completedCases, loading, error, lastSync, refresh: load, updateStatus }
}
/**
 * src/hooks/useAirstrips.js
 * -------------------------
 * Loads Kenya airstrips (with lat/lng) from the database via API.
 * Fetched once on mount; refreshed only on explicit call.
 */

import { useState, useEffect } from 'react'
import { fetchAirstrips } from '../api/client'

export function useAirstrips() {
  const [airstrips, setAirstrips] = useState([])
  const [loading, setLoading]     = useState(true)
  const [error, setError]         = useState(null)

  useEffect(() => {
    fetchAirstrips()
      .then(data => {
        // Only map-able airstrips (must have coordinates)
        setAirstrips(data.filter(a => a.latitude != null && a.longitude != null))
      })
      .catch(err => setError(err.message))
      .finally(() => setLoading(false))
  }, [])

  return { airstrips, loading, error }
}

const STATUS_TRANSITIONS = {
  RECEIVED:   ['DISPATCHED', 'CANCELLED'],
  DISPATCHED: ['AIRBORNE',   'CANCELLED'],
  AIRBORNE:   ['LANDED',     'CANCELLED'],
  LANDED:     ['COMPLETED'],
  COMPLETED:  [],
  CANCELLED:  [],
}

  const [eta, setEta]                 = useState(c?.eta_minutes ?? '')

  const [error, setError]             = useState(null)

    setNotes(c?.notes ?? '')
    setEta(c?.eta_minutes ?? '')
    setNextStatus('')
    setError(null)
  }, [c?.id])

  useEffect(() => {
    const handler = e => { if (e.key === 'Escape') onClose() }
    window.addEventListener('keydown', handler)

  }, [onClose])

    setError(null)

      await onUpdate(c.id, nextStatus, notes, eta ? Number(eta) : null)
      onClose()
    } catch (err) {
      setError(err.message)
    } finally {
      setSubmitting(false)
    }
  }

  const created = c.created_at
    ? new Date(c.created_at).toLocaleString('en-KE', {
        weekday: 'short', month: 'short', day: 'numeric',
        hour: '2-digit', minute: '2-digit',
      })
    : '—'

  return (
    <>
      {/* Backdrop */}
      <div
        onClick={onClose}
        style={{
          position: 'fixed', inset: 0,
          background: 'rgba(0,0,0,0.6)',

          zIndex: 100,

          borderBottom: '1px solid var(--border)',
          display: 'flex',
          alignItems: 'flex-start',
          justifyContent: 'space-between',
          flexShrink: 0,
        }}>
          <div>
            <div style={{
              fontFamily: "'Space Mono', monospace",
              fontSize: '1rem',
              fontWeight: 700,
              color: 'var(--accent)',
              marginBottom: 6,
            }}>
              {c.id}
            </div>
            <StatusBadge status={c.status} size="lg" />
          </div>
          <button
            onClick={onClose}
            style={{
              background: 'transparent',
              border: '1px solid var(--border)',
              borderRadius: 4,
              color: 'var(--text-muted)',
              fontSize: '1rem',
              cursor: 'pointer',
              padding: '2px 8px',
            }}
          >
            ✕
          </button>
        </div>

        {/* Body */}
        <div style={{ flex: 1, overflowY: 'auto', padding: '16px 20px' }}>

          <Field label="Airstrip" value={c.airstrip_name} />

          <Field label="Created" value={created} />
          {c.notes && <Field label="Notes" value={c.notes} />}

          {/* Divider */}
          <div style={{ borderTop: '1px solid var(--border)', margin: '20px 0' }} />

          {/* Status update form */}
          {transitions.length > 0 && (
            <>
              <Label>Update Status</Label>

                  background: 'rgba(239,68,68,0.1)',

        </div>
      </div>
    </>
  )

const STATUS_ORDER = ['RECEIVED', 'DISPATCHED', 'AIRBORNE', 'LANDED', 'COMPLETED']

}

export default function StatsBar({ cases }) {
  // Count per status
  const counts = STATUS_ORDER.reduce((acc, s) => {
    acc[s] = cases.filter(c => c.status === s).length
    return acc
  }, {})

  const active = cases.filter(c => !['COMPLETED', 'CANCELLED'].includes(c.status)).length
  const avgEta = (() => {
    const withEta = cases.filter(c => c.eta_minutes && !['COMPLETED', 'CANCELLED'].includes(c.status))

    return Math.round(withEta.reduce((s, c) => s + c.eta_minutes, 0) / withEta.length)

    <div style={{
      display: 'grid',
      gridTemplateColumns: 'repeat(auto-fit, minmax(120px, 1fr))',

      {STATUS_ORDER.slice(0, 4).map(s => (
        <StatCard key={s} label={s[0] + s.slice(1).toLowerCase()} value={counts[s]} accent={STATUS_COLORS[s]} />

/**
 * src/api/client.js
 * -----------------
 * Thin typed wrappers around every AeroCure backend API endpoint.
 * Base URL is configured here — swap for production when going live.
 *
 * During dev:  Vite proxies /api/* → http://localhost:8000
 * Production:  set VITE_API_BASE in .env.production
 */

const BASE = import.meta.env.VITE_API_BASE ?? ''

async function request(path, options = {}) {
  const res = await fetch(`${BASE}${path}`, {
    headers: { 'Content-Type': 'application/json', ...options.headers },
    ...options,
  })
  if (!res.ok) {
    const text = await res.text().catch(() => res.statusText)
    throw new Error(`API ${options.method ?? 'GET'} ${path} → ${res.status}: ${text}`)
  }
  return res.json()
}
